### Clean the data in snp.csv

``` r
snp_df = 
  read_csv("data/snp.csv") |>
  janitor::clean_names() |>
  mutate(
    date = as.Date(date, format = "%m/%d/%y"),
    date = as.Date(ifelse(date > Sys.Date(),
                           format(date, "19%y-%m-%d"),
                           format(date)))
  ) |>
  separate(date, into = c("year", "month", "day"), sep = "-") |>
  mutate(
    year = as.numeric(year)
```
